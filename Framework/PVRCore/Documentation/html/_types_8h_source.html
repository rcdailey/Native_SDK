413 \return The size of the Datatype in bytes.

415 inline uint32 size(Enum type)
416 {
417  switch (type)
418  {

421  return 0;
422  case DataType::Float32:
423  return static_cast<uint32>(sizeof(float));
424  case DataType::Int32:

426  return static_cast<uint32>(sizeof(int));

429  case DataType::UInt16:
430  return static_cast<uint32>(sizeof(unsigned short));
431  case DataType::RGBA:
432  return static_cast<uint32>(sizeof(unsigned int));
433  case DataType::ABGR:
434  return static_cast<uint32>(sizeof(unsigned int));
435  case DataType::ARGB:
436  return static_cast<uint32>(sizeof(unsigned int));
437  case DataType::D3DCOLOR:
438  return static_cast<uint32>(sizeof(unsigned int));
439  case DataType::UBYTE4:

491 /*!*********************************************************************************************
492 \brief Return if the format is Normalized (represents a range between 0..1 for unsigned types
493  or between -1..1 for signed types)
494 \param type The format to test.
495 \return True if the format is Normalised.
496 \description A Normalised format is a value that is stored as an Integer, but that actually
497 represents a value from 0..1 or -1..1 instead of the numeric value
498 of the integer. For example, for a normalised unsigned byte value, the value
499 0 represents 0.0, the value 127 represents 0.5 and the value 255 represents 1.0.
500 ***********************************************************************************************/
501 inline bool isNormalised(Enum type)
502 {
503  return (type == DataType::Int8Norm || type == DataType::UInt8Norm
504  || type == DataType::Int16Norm
505  || type == DataType::UInt16Norm);
506 }
