149  {

151  Api::getApiName(m_apiType));
152  return Result::UnsupportedRequest;

155 #if BUILD_API_MAX&&BUILD_API_MAX<30

159  " but an ES 3.0 context was requested.");
160  return Result::UnsupportedRequest;

162 #endif
163 #if BUILD_API_MAX&&BUILD_API_MAX<31
164  if (m_apiType >= Api::OpenGLES31)

166  Log(Log.Critical,

201  void setUpCapabilities()
202  {
203  ApiCapabilitiesPrivate& caps = (ApiCapabilitiesPrivate&)m_apiCapabilities;
204  caps.maxglslesversion = (m_apiType >= Api::OpenGLES31 ? 310 : m_apiType >= Api::OpenGLES3 ? 300 : 200);
205 
206  // EXTENSIONS -- SEE TOP OF THIS FILE.
207  // For each extension, make sure that the we properly determine native or extension support.
208  for (int i = 0; i < sizeof(extensionMap) / sizeof(extensionMap[0]); ++i)
209  {
210  if (extensionMap[i].minCoreLevel != Api::Unspecified && m_apiType >= extensionMap[i].minCoreLevel)
211  {
212  caps.nativeSupport[extensionMap[i].capability] = true;
213  }
214  else if (extensionMap[i].minExtensionLevel != Api::Unspecified && m_apiType >= extensionMap[i].minExtensionLevel)
215  {
216  caps.extensionSupport[extensionMap[i].capability] = isExtensionSupported(extensionMap[i].extensionString);
217  }

421  void enableAttribute(uint16 attributeIdx)
422  {
423  m_renderStatesTracker.attributesToEnableBitfield |= (1 << attributeIdx);
424  if ((m_renderStatesTracker.attributesEnabledBitfield & (1 << attributeIdx)) == 0)
425  {
426  gl::EnableVertexAttribArray(attributeIdx);
427  m_renderStatesTracker.attributesEnabledBitfield |= (1 << attributeIdx);
428  }
429  m_renderStatesTracker.attributesMaxToEnable = (std::max)(attributeIdx, m_renderStatesTracker.attributesMaxToEnable) + 1;
430  }

432  /*!*********************************************************************************************************************
433  \brief Internal use. State tracking.disable unneeded vertex attributes.
434  ***********************************************************************************************************************/
435  void disableUnneededAttributes()
436  {
437  uint32 deactivate = m_renderStatesTracker.attributesEnabledBitfield ^ m_renderStatesTracker.attributesToEnableBitfield;
438 
439  for (uint8 i = 0; i < m_renderStatesTracker.attributesMaxEnabled; ++i)
440  {
441  if (Bitfield<uint32>::isSet(deactivate, i))
442  {
443  gl::DisableVertexAttribArray(i);
444  }
445  }
446  m_renderStatesTracker.attributesEnabledBitfield = m_renderStatesTracker.attributesToEnableBitfield;
447  m_renderStatesTracker.attributesMaxEnabled = m_renderStatesTracker.attributesMaxToEnable;
448  m_renderStatesTracker.attributesMaxToEnable = 0;
449  m_renderStatesTracker.attributesToEnableBitfield = 0;
450  }
