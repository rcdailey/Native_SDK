189 /*!****************************************************************************************************************
190 \brief API graphics pipeline wrapper. A GraphicsPipeline represents the configuration of almost the entire RenderState,
191 including vertex description, primitive assembly, Shader configuration, rasterization, blending etc. Access through
192 the Framework managed object GraphicsPipeline.
193 A ParentableGraphicsPipeline is a pipeline that is suitable to function as the "Parent" of another pipeline, helping
194 to create efficient Pipeline Hierarchies.
195 \description ParentableGraphicsPipelines can and should be used to make switching between different pipelines more
196 efficient. In effect, a ParentableGraphicsPipeline allows the user to create another (non-parentable pipeline) as
197 a "diff" of the state between the Parentable pipeline and itself, making the transition between them very efficient.
198 ******************************************************************************************************************/

240 inline GraphicsPipelineImpl::PipelineRelation::Enum GraphicsPipelineImpl::getRelation(
241  GraphicsPipelineImpl* first, GraphicsPipelineImpl* second)
242 {
243  if (first)
244  {
245  if (second)
246  {
247  return first == second ? PipelineRelation::Identity :
248  first->m_parent == second ? PipelineRelation::Child_Father :
249  first->m_parent == second->m_parent ? first->m_parent == NULL ? PipelineRelation::Unrelated : PipelineRelation::Siblings :
250  first == second->m_parent ? PipelineRelation::Father_Child :
251  PipelineRelation::Unrelated;
252  }
253  else { return PipelineRelation::NotNull_Null; }
254  }
255  else { return second ? PipelineRelation::Null_NotNull : PipelineRelation::Null_Null; }
256 }
